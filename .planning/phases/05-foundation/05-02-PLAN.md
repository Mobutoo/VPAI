```bash
#!/bin/bash
# {{ ansible_managed }}
# NocoDB Content Factory table provisioning — idempotent
# Creates tables: brands, contents, scenes + seeds Paul Taff brand

set -euo pipefail

BASE_URL="{{ content_factory_nocodb_base_url }}"
TOKEN="{{ nocodb_api_token }}"
SENTINEL="{{ content_factory_config_dir }}/.nocodb-provision-complete"

if [ -f "$SENTINEL" ]; then
  echo "NocoDB tables already provisioned (sentinel exists). Skipping."
  exit 0
fi

echo "[$(date)] Starting NocoDB Content Factory provisioning..."

# Step 1: Find or create the content-factory base
echo "[$(date)] Looking for content-factory base..."
BASES=$(curl -sf -H "xc-token: ${TOKEN}" "${BASE_URL}/api/v2/meta/bases/")
BASE_ID=$(echo "$BASES" | python3 -c "
import sys, json
data = json.load(sys.stdin)
for b in data.get('list', []):
    if b['title'] == '{{ content_factory_nocodb_base_name }}':
        print(b['id'])
        break
" 2>/dev/null || true)

if [ -z "$BASE_ID" ]; then
  echo "[$(date)] Creating content-factory base..."
  CREATE_RESULT=$(curl -sf -X POST \
    -H "xc-token: ${TOKEN}" \
    -H "Content-Type: application/json" \
    "${BASE_URL}/api/v2/meta/bases/" \
    -d '{"title": "{{ content_factory_nocodb_base_name }}", "type": "database"}')
  BASE_ID=$(echo "$CREATE_RESULT" | python3 -c "import sys, json; print(json.load(sys.stdin)['id'])")
  echo "[$(date)] Base created: $BASE_ID"
else
  echo "[$(date)] Base found: $BASE_ID"
fi

# Step 2: List existing tables
TABLES=$(curl -sf -H "xc-token: ${TOKEN}" "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables")

table_exists() {
  echo "$TABLES" | python3 -c "
import sys, json
data = json.load(sys.stdin)
for t in data.get('list', []):
    if t['title'] == '$1':
        print(t['id'])
        break
" 2>/dev/null || true
}

# Step 3: Create brands table
BRANDS_ID=$(table_exists "brands")
if [ -z "$BRANDS_ID" ]; then
  echo "[$(date)] Creating brands table..."
  BRANDS_RESULT=$(curl -sf -X POST \
    -H "xc-token: ${TOKEN}" \
    -H "Content-Type: application/json" \
    "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables" \
    -d '{
      "title": "brands",
      "columns": [
        {"title": "name", "uidt": "SingleLineText"},
        {"title": "tagline", "uidt": "SingleLineText"},
        {"title": "tone", "uidt": "LongText"},
        {"title": "palette", "uidt": "JSON"},
        {"title": "typography", "uidt": "SingleLineText"},
        {"title": "target_audience", "uidt": "SingleLineText"},
        {"title": "platforms", "uidt": "JSON"}
      ]
    }')
  BRANDS_ID=$(echo "$BRANDS_RESULT" | python3 -c "import sys, json; print(json.load(sys.stdin)['id'])")
  echo "[$(date)] brands table created: $BRANDS_ID"
else
  echo "[$(date)] brands table exists: $BRANDS_ID"
fi

# Step 4: Create contents table
CONTENTS_ID=$(table_exists "contents")
if [ -z "$CONTENTS_ID" ]; then
  echo "[$(date)] Creating contents table..."
  CONTENTS_RESULT=$(curl -sf -X POST \
    -H "xc-token: ${TOKEN}" \
    -H "Content-Type: application/json" \
    "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables" \
    -d '{
      "title": "contents",
      "columns": [
        {"title": "brand_id", "uidt": "SingleLineText"},
        {"title": "kitsu_project_id", "uidt": "SingleLineText"},
        {"title": "title", "uidt": "SingleLineText"},
        {"title": "format", "uidt": "SingleSelect", "dtxp": "reel,carousel,post,story,ad"},
        {"title": "status", "uidt": "SingleSelect", "dtxp": "brief,concept,script,storyboard,sound,assets,rough,fine,review,adapt,published"},
        {"title": "current_phase", "uidt": "Number"},
        {"title": "current_step", "uidt": "Number"},
        {"title": "brief", "uidt": "JSON"},
        {"title": "script", "uidt": "LongText"},
        {"title": "storyboard", "uidt": "JSON"},
        {"title": "assets_urls", "uidt": "JSON"},
        {"title": "final_url", "uidt": "URL"},
        {"title": "published_at", "uidt": "DateTime"},
        {"title": "instagram_id", "uidt": "SingleLineText"}
      ]
    }')
  CONTENTS_ID=$(echo "$CONTENTS_RESULT" | python3 -c "import sys, json; print(json.load(sys.stdin)['id'])")
  echo "[$(date)] contents table created: $CONTENTS_ID"
else
  echo "[$(date)] contents table exists: $CONTENTS_ID"
fi

# Step 5: Create scenes table
SCENES_ID=$(table_exists "scenes")
if [ -z "$SCENES_ID" ]; then
  echo "[$(date)] Creating scenes table..."
  SCENES_RESULT=$(curl -sf -X POST \
    -H "xc-token: ${TOKEN}" \
    -H "Content-Type: application/json" \
    "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables" \
    -d '{
      "title": "scenes",
      "columns": [
        {"title": "content_id", "uidt": "SingleLineText"},
        {"title": "scene_number", "uidt": "Number"},
        {"title": "description", "uidt": "LongText"},
        {"title": "dialogue", "uidt": "LongText"},
        {"title": "screen_text", "uidt": "SingleLineText"},
        {"title": "duration_sec", "uidt": "Decimal"},
        {"title": "transition", "uidt": "SingleLineText"},
        {"title": "visual_type", "uidt": "SingleSelect", "dtxp": "motion_design,ai_generative,stock,meme"},
        {"title": "provider", "uidt": "SingleSelect", "dtxp": "remotion,comfyui,fal_ai,seedance,seedream"},
        {"title": "asset_url", "uidt": "URL"},
        {"title": "version", "uidt": "Number"},
        {"title": "status", "uidt": "SingleSelect", "dtxp": "draft,validated,invalidated"}
      ]
    }')
  SCENES_ID=$(echo "$SCENES_RESULT" | python3 -c "import sys, json; print(json.load(sys.stdin)['id'])")
  echo "[$(date)] scenes table created: $SCENES_ID"
else
  echo "[$(date)] scenes table exists: $SCENES_ID"
fi

# Step 6: Seed Paul Taff brand profile
echo "[$(date)] Checking Paul Taff brand profile..."
BRAND_ROWS=$(curl -sf -H "xc-token: ${TOKEN}" \
  "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables/${BRANDS_ID}/records?where=(name,eq,Paul Taff)&limit=1" 2>/dev/null || true)
BRAND_EXISTS=$(echo "$BRAND_ROWS" | python3 -c "
import sys, json
data = json.load(sys.stdin)
print(len(data.get('list', [])))
" 2>/dev/null || echo "0")

if [ "$BRAND_EXISTS" = "0" ]; then
  echo "[$(date)] Creating Paul Taff brand profile..."
  curl -sf -X POST \
    -H "xc-token: ${TOKEN}" \
    -H "Content-Type: application/json" \
    "${BASE_URL}/api/v2/meta/bases/${BASE_ID}/tables/${BRANDS_ID}/records" \
    -d '{
      "name": "Paul Taff",
      "tagline": "Des agents IA avec du caractere",
      "tone": "Decale, sarcastique, audacieux. Flash Studio = quand tes agents IA ont plus de personnalite que ton ex.",
      "palette": "{\"primary\": \"#FF6B35\", \"secondary\": \"#004E89\", \"accent\": \"#F7C948\", \"dark\": \"#1A1A2E\", \"light\": \"#FAFAFA\"}",
      "typography": "Inter (headings), JetBrains Mono (code/tech)",
      "target_audience": "Dev/founders 25-40, early adopters, tech-savvy creators",
      "platforms": "[\"instagram\"]"
    }'
  echo "[$(date)] Paul Taff brand created"
else
  echo "[$(date)] Paul Taff brand already exists"
fi

# Write sentinel
touch "$SENTINEL"
echo "[$(date)] NocoDB Content Factory provisioning complete."
```

**1. roles/content-factory-provision/templates/provision-qdrant.sh.j2:**
```bash
#!/bin/bash
# {{ ansible_managed }}
# Qdrant brand-voice collection provisioning — idempotent
# Creates collection + seeds initial Paul Taff brand vectors via LiteLLM embeddings

set -euo pipefail

QDRANT_URL="{{ content_factory_qdrant_base_url }}"
QDRANT_KEY="{{ qdrant_api_key }}"
LITELLM_URL="{{ content_factory_litellm_base_url }}"
LITELLM_KEY="{{ litellm_master_key }}"
COLLECTION="{{ content_factory_qdrant_collection }}"
VECTOR_SIZE="{{ content_factory_qdrant_vector_size }}"
DISTANCE="{{ content_factory_qdrant_distance }}"
SENTINEL="{{ content_factory_config_dir }}/.qdrant-provision-complete"
EMBEDDING_MODEL="{{ content_factory_embedding_model }}"

if [ -f "$SENTINEL" ]; then
  echo "Qdrant brand-voice already provisioned (sentinel exists). Skipping."
  exit 0
fi

echo "[$(date)] Starting Qdrant brand-voice provisioning..."

# Step 1: Create collection (PUT is idempotent in Qdrant)
echo "[$(date)] Creating/verifying brand-voice collection..."
HTTP_CODE=$(curl -sf -o /dev/null -w '%{http_code}' -X PUT \
  -H "api-key: ${QDRANT_KEY}" \
  -H "Content-Type: application/json" \
  "${QDRANT_URL}/collections/${COLLECTION}" \
  -d "{\"vectors\": {\"size\": ${VECTOR_SIZE}, \"distance\": \"${DISTANCE}\"}}" \
  2>/dev/null || echo "000")

if [ "$HTTP_CODE" = "200" ] || [ "$HTTP_CODE" = "409" ]; then
  echo "[$(date)] Collection ${COLLECTION} ready (HTTP ${HTTP_CODE})"
else
  echo "[$(date)] ERROR: Failed to create collection (HTTP ${HTTP_CODE})"
  exit 1
fi

# Step 2: Check if seed vectors already exist
POINT_COUNT=$(curl -sf \
  -H "api-key: ${QDRANT_KEY}" \
  "${QDRANT_URL}/collections/${COLLECTION}" \
  | python3 -c "import sys, json; print(json.load(sys.stdin).get('result', {}).get('points_count', 0))" 2>/dev/null || echo "0")

if [ "$POINT_COUNT" -gt "0" ]; then
  echo "[$(date)] Collection already has ${POINT_COUNT} vectors. Skipping seed."
  touch "$SENTINEL"
  echo "[$(date)] Qdrant provisioning complete."
  exit 0
fi

# Step 3: Generate embeddings via LiteLLM proxy
# Seed texts: Paul Taff brand description and tone guidelines
SEED_TEXTS=(
  "Paul Taff - Flash Studio. Des agents IA avec du caractere. Tone decale, sarcastique, audacieux. Quand tes agents IA ont plus de personnalite que ton ex. Colors: orange vif #FF6B35, bleu profond #004E89, jaune accent #F7C948."
  "Style Instagram Paul Taff: reels motion-text energiques, memes tech sarcastiques, showcases produit avec overlays audacieux. Target: dev et founders 25-40 ans, early adopters. Voix: tutoyement, argot tech, punchlines courtes, humour noir leger."
  "Brand guidelines Paul Taff Flash Studio: typographie Inter pour les titres, JetBrains Mono pour le code. Transitions rapides, rythme soutenu. Pas de bullshit corporate, pas de jargon marketing creux. On parle comme un dev qui a trop de cafeINe et une deadline dans 2h."
)

echo "[$(date)] Generating embeddings via LiteLLM..."
for i in "${!SEED_TEXTS[@]}"; do
  POINT_ID=$((i + 1))
  TEXT="${SEED_TEXTS[$i]}"

  # Get embedding from LiteLLM
  EMBEDDING_RESPONSE=$(curl -sf -X POST \
    -H "Authorization: Bearer ${LITELLM_KEY}" \
    -H "Content-Type: application/json" \
    "${LITELLM_URL}/embeddings" \
    -d "{\"model\": \"${EMBEDDING_MODEL}\", \"input\": \"${TEXT}\"}")

  EMBEDDING=$(echo "$EMBEDDING_RESPONSE" | python3 -c "
import sys, json
data = json.load(sys.stdin)
print(json.dumps(data['data'][0]['embedding']))
")

  # Determine content_type based on index
  case $i in
    0) CONTENT_TYPE="brand_description" ;;
    1) CONTENT_TYPE="style_guide" ;;
    2) CONTENT_TYPE="brand_guidelines" ;;
  esac

  # Upsert point to Qdrant
  curl -sf -X PUT \
    -H "api-key: ${QDRANT_KEY}" \
    -H "Content-Type: application/json" \
    "${QDRANT_URL}/collections/${COLLECTION}/points" \
    -d "{
      \"points\": [{
        \"id\": ${POINT_ID},
        \"vector\": ${EMBEDDING},
```
