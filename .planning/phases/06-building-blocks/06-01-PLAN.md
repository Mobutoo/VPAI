```tsx
import React from "react";
import { AbsoluteFill, spring, useCurrentFrame, useVideoConfig } from "remotion";

interface HelloWorldProps {
  titleText: string;
  titleColor?: string;
}

export const HelloWorld: React.FC<HelloWorldProps> = ({ titleText, titleColor = "#ffffff" }) => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();
  const opacity = spring({ fps, frame, config: { damping: 200 } });
  return (
    <AbsoluteFill style={{ backgroundColor: "#0f0f0f", justifyContent: "center", alignItems: "center" }}>
      <div style={{ fontSize: 80, fontWeight: "bold", color: titleColor, opacity, fontFamily: "sans-serif", textAlign: "center", padding: "0 40px" }}>
        {titleText}
      </div>
    </AbsoluteFill>
  );
};
```

**Step 1: Create `roles/remotion/files/remotion/types.ts`** with these exact exported interfaces:

```typescript
export interface SceneData {
  scene_number: number;
  description: string;
  dialogue?: string;
  screen_text?: string;
  duration_sec: number;
  transition: "cut" | "fade" | "zoom" | "swipe" | "blur";
  visual_type: "motion_design" | "ai_generative" | "stock" | "meme";
  asset_url?: string;
}

export interface BrandProfile {
  name: string;
  palette: { primary: string; accent: string; background?: string };
  typography: { heading: string; body: string };
  logo_url?: string;
}

export interface AudioConfig {
  url?: string;
  startFrom?: number;
  volume?: number;
}

export interface ReelProps {
  scenes: SceneData[];
  brand: BrandProfile;
  audio?: AudioConfig;
}
```

**Step 2: Create `ReelMotionText/ReelMotionText.tsx`** -- Animated text over gradient backgrounds.

The composition implements `ReelProps`. For each scene:
- Background: CSS `linear-gradient` using `brand.palette.primary` and `brand.palette.accent` (angle varies per scene: `${scene.scene_number * 45}deg`)
- Text animation: use `spring()` for opacity + scale on `scene.screen_text` (fallback to `scene.description` if no screen_text)
- Font: `brand.typography.heading` for main text, `brand.typography.body` for dialogue/description
- Transition between scenes: simple opacity crossfade via `interpolate()` in the last 5 frames of each Sequence
- If `scene.asset_url` is provided, render as `<Img>` background (from `remotion` package) behind the text
- Include `<Audio>` component if `props.audio?.url` is provided

Calculate `durationInFrames` per scene as `Math.ceil(scene.duration_sec * fps)`. Use `<Sequence>` to place each scene at its computed frame offset.

Style: centered text layout, `fontSize: 72` for heading, `fontSize: 36` for body, `padding: 60px`, white text with subtle text-shadow (`0 2px 20px rgba(0,0,0,0.5)`).

**Step 3: Create `ReelMemeSkit/ReelMemeSkit.tsx`** -- Meme/skit format with image layers and text overlays.

Implements `ReelProps`. For each scene:
- If `scene.asset_url`: render full-bleed background image/video placeholder using `<Img>` from remotion
- If no asset: solid color background from `brand.palette.background` (fallback: `#1a1a2e`)
- Top third: `scene.screen_text` in large bold white text (meme caption style, `fontSize: 56`, `textTransform: "uppercase"`, black outline `textShadow: "-2px -2px 0 #000, 2px -2px 0 #000, -2px 2px 0 #000, 2px 2px 0 #000"`)
- Bottom third: `scene.dialogue` in smaller text (`fontSize: 32`), white on semi-transparent black bar (`rgba(0,0,0,0.7)`, `padding: 20px`)
- Scene entrance: `spring()` scale from 1.1 to 1.0 (Ken Burns zoom-out effect)
- Transition: hard cut between scenes (no crossfade -- meme style)

**Step 4: Create `ReelFeatureShowcase/ReelFeatureShowcase.tsx`** -- Product demo with overlays.

Implements `ReelProps`. For each scene:
- Main area (top 75%): `scene.asset_url` as `<Img>` if available, or gradient placeholder with `scene.description` text
- Bottom overlay strip (bottom 25%): semi-transparent brand-colored bar (`brand.palette.primary` at 0.85 opacity)
  - `scene.screen_text` in `brand.typography.heading` font, white, `fontSize: 40`
  - Small brand name badge top-right corner: `brand.name` in `brand.palette.accent` background, `fontSize: 18`, `borderRadius: 8px`, `padding: 6px 16px`
- Animations: overlay slides up from bottom using `interpolate(frame, [startFrame, startFrame+10], [100, 0])` for translateY
- If first scene: add a 1-second intro with brand name centered large (`fontSize: 96`) with `spring()` scale-in
- If last scene: add CTA text "Swipe up" or `scene.screen_text` pulsing with `Math.sin(frame * 0.1) * 0.1 + 1` scale

**Step 5: Create `ReelTeaser/ReelTeaser.tsx`** -- Fast cuts, blur reveals, CTA.

Implements `ReelProps`. This is a short-form (15s typical) composition:
- First scene: only show blurred version of `scene.description` text (CSS `filter: blur(8px)`) then reveal with `interpolate()` from `blur(8px)` to `blur(0px)` over 15 frames
- Middle scenes: rapid alternation -- each scene gets hard cut, text appears with `spring()` from bottom (`translateY` from 50 to 0), stays briefly, cuts
- Background: dark (`#0a0a0a`) with accent color flash frames (2-frame flash of `brand.palette.accent` at scene transitions)
- Last scene: CTA with `brand.palette.primary` background, large centered text (`fontSize: 80`), `spring()` scale entrance
- Text style: bold, uppercase, `brand.typography.heading`, white, `letterSpacing: 4`

All compositions: use `React.FC<ReelProps>` type, import types from `../types`, use `useCurrentFrame()` and `useVideoConfig()` hooks, and wrap everything in `<AbsoluteFill>`.

Do NOT use `calculateMetadata` (not needed for this version -- default `durationInFrames` will be set in Root.tsx and can be overridden via `inputProps` at render time).
